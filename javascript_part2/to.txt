var str = "border-bottom"; /// borderBottom



function capilzie(text) {
    var arr = text.split('-');
    for(var i = 1; i < arr.length; i++){
        arr[i] = arr[i].toUpperCase().charAt(0) + arr[i].slice(1);
    }
    arr[1].charAt(0).toUpperCase();

    return arr.join("");
}

console.log(capilzie(str));
